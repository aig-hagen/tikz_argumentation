\documentclass[]{article}
\usepackage{argumentation}

\begin{document}
\begin{af}
	\argument[](a1){$a_{1}$} at (0.00, -0.00)
	\argument[](a2){$a_{2}$} at (1.75, -0.00)
	\argument[](a3){$a_{3}$} at (3.50, -0.00)
	\argument[](a4){$a_{4}$} at (0.00, -1.75)
	\argument[](a5){$a_{5}$} at (1.75, -1.75)
	\argument[](a6){$a_{6}$} at (3.50, -1.75)
	\annotation[yshift=-0.3cm]{a1}{$a_{2}$}
	\annotation[yshift=-0.3cm]{a2}{$\neg a_{5}$}
	\annotation[yshift=-0.3cm]{a3}{$\neg a_{2}$}
	\annotation[yshift=-1.7cm]{a4}{$\neg a_{5}$}
	\annotation[yshift=-1.7cm]{a5}{$\neg a_{2} \land a_{4} \land \neg a_{6}$}
	\annotation[yshift=-1.7cm]{a6}{$\bot$}
	\attack[]{a2}{a1}
	\attack[]{a6}{a5}
	\attack[bend right]{a5}{a2}
	\attack[bend right]{a4}{a5}
	\attack[bend right]{a5}{a4}
	\attack[bend right]{a2}{a5}
	\attack[]{a2}{a3}
\end{af}
\end{document}
